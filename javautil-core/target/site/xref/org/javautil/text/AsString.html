10  /**
11   * Simple way to get an object formatted as a String.
12   * 
13   * Transients are by default not emitted but don't use the transient qualifier
14   * as a means to suppress if you plan on using HashCodeBuilder or EqualsBuilder
15   * to build a hash and the field is actually part of the identifier.
16   * 
17   * 
18   * Configuration is primarily through the flags see
19   * 
20   * @see ToStringStyleFlags
21   * @see ConfigurableToStringStyle
22   * 
23   *      Sample usage
24   * 
25   *      <pre>
26   * <code>
27   * AsString as = new AsString(DONT_USE_IDENTITY_HASH_CODE,SUPPRESS_TIME);
28   * logger.debug(as.toString(object);
29   * </code>
30   * </pre>
31   * 
32   *      todo document default values for Configurable
33   * 
34   * @author jjs at dbexperts dot com
35   */

37  public class AsString {
38  	private ConfigurableToStringStyle style = new ConfigurableToStringStyle(
39  			DONT_EMIT_IDENTITY_HASH_CODE, USE_SHORT_CLASS_NAME,
40  			DONT_EMIT_TRANSIENTS, DONT_EMIT_STATICS);

45  	/**
46  	 * Fields to be suppressed, use for passwords, etc.
47  	 */
48  	private String[] excludeFieldNames;
49  
50  	public AsString(final ToStringStyleFlags... flags) {
51  		style.setFlags(flags);
52  
53  	}
54  
55  	public String toString(final Object object) {
56  		String string = null;
57  		// TODO should not instantiate every time
58  		if (excludeFieldNames != null) {
59  			final ReflectionToStringBuilder localBuilder = new ReflectionToStringBuilder(
60  					object, style, null, null, style.getEmitTransients(),
61  					style.getEmitStatics());
62  			localBuilder.setExcludeFieldNames(excludeFieldNames);
63  			string = localBuilder.toString();
64  		} else {
65  			string = ReflectionToStringBuilder.toString(object, style,
66  					style.getEmitTransients(), style.getEmitStatics());
67  		}
68  		return string;
69  
70  	}
71  
72  	public ConfigurableToStringStyle getStyle() {
73  		return style;
74  	}
75  
76  	public void setStyle(final ConfigurableToStringStyle style) {
77  		this.style = style;
78  	}
