220 	public void appendRow(final List values) {
221 		if (footerValues != null) {
222 			throw new IllegalStateException(
223 					"Please add all columns before computing footers.");
224 		}
225 		List<T> newValues;
226 
227 		if (values == null) {
228 			throw new IllegalArgumentException("values is null");
229 		}
230 		if (values.size() < metadata.getColumnCount()) {
231 			newValues = new ArrayList<T>(metadata.getColumnCount());
232 			logger.warn("values size " + values.size()
233 					+ " but meta column count " + metadata.getColumnCount());
234 			while (values.size() < metadata.getColumnCount()) {
235 				newValues.add(null);
236 			}
237 
238 		} else if (values.size() > metadata.getColumnCount()) {
239 			newValues = values.subList(0, metadata.getColumnCount() - 1);
240 		} else if (values.size() == metadata.getColumnCount()) {
241 			newValues = values;
242 		} else {
243 			throw new IllegalStateException("logic error");
244 		}
245 		validateRow(newValues);
246 		cells.add(newValues);
247 	}

249 	private void validateRow(final List<T> colData) {
250 
251 		if (colData.size() != metadata.getColumnCount()) {
252 			throw new IllegalStateException("row size " + colData.size()
253 					+ " meta cols " + metadata.getColumnCount());
254 		}
255 		for (int i = 0; i < colData.size(); i++) {
256 			final DataType dt = metadata.getColumnType(i);
257 			final Object o = colData.get(i);
258 			if (!dt.isType(o) && coerce(o, dt) != null) {
259 				throw new IllegalArgumentException("column at index " + i
260 						+ " not valid for " + dt);
261 			}
262 		}
263 	}

268 		Object returnValue = null;
269 		if (type == null) {
270 			throw new IllegalArgumentException("type is null");
271 		}
272 		if (type == DataType.INTEGER && o != null) {
273

275 			if (o instanceof String) {
276 				final String string = (String) o;
277 				try {
278 					final Integer x = Integer.parseInt(string);
279 					returnValue = x;
280 				} catch (final NumberFormatException nfe) {
281 					throw new IllegalArgumentException(" string " + string
282 							+ "cannot be parsed as an integer", nfe);

384 	@Override
385 	public void addFooter(final int columnIndex,
386 			final CollectionMathOperation footerMathOper) {
387 		final Number[] columnValues = new Number[cells.size()];
388 		for (int index = 0; index < cells.size(); index++) {
389 			final List<T> cellRow = cells.get(index);
390 			final Object object = cellRow.get(columnIndex);
391 			if (object != null && !(object instanceof Number)) {
392 				throw new IllegalArgumentException(
393 						"columnIndex references a non java.lang.Number value.");
394 			}
395 			final Number number = (Number) object;
396 			columnValues[index] = number;
397 		}
398 
399 		if (footerValues == null) {
400 			footerValues = new Object[metadata.getColumnCount()];
401 		} else if (columnIndex >= footerValues.length) {
402 			throw new IllegalStateException(
403 					"Please add all columns before computing footers.");
404 		}
405 
406 		final double computedValue = footerMathOper.compute(columnValues);
407 		footerValues[columnIndex] = computedValue;
408 	}

429 	public void applyFilters(final MutableDatasetFilter... filters) {
430 		for (final MutableDatasetFilter filter : filters) {
431 			final Iterator<List<T>> iter = cells.iterator();
432 			while (iter.hasNext()) {
433 				final List<T> row = iter.next();
434 				final String columnName = filter.getColumnName();
435 				final int columnIndex = metadata.getColumnIndex(columnName);
436 				final Object value = row.get(columnIndex);
437 				if (!filter.isMatch(value)) {
438 					iter.remove();
439 				}
440 			}
441 		}
442 	}
