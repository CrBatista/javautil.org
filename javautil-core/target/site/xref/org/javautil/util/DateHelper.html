267 	/**
268 	 * Please note that the 4th of October, 1582 is directly followed by the
269 	 * 15th of October 1582 as decreed by Pope Gregor. (Pope Gregor forced this
270 	 * in order to achieve better agreement between the civil and the
271 	 * astronomical calendar)
272 	 */
273 	public static double julianDayNumber(final java.util.Date dt) {
274 		final GregorianCalendar cal = new GregorianCalendar();
275 		cal.setTime(dt);
276 		int year = cal.get(java.util.Calendar.YEAR) + 1900;
277 		int month = cal.get(java.util.Calendar.MONTH) + 1;
278 		final int dayOfMonth = cal.get(java.util.Calendar.DATE);
279 		final int hour = cal.get(java.util.Calendar.HOUR);
280 		final int minutes = cal.get(java.util.Calendar.MINUTE);
281 		final int seconds = cal.get(java.util.Calendar.SECOND);
282 		//
283 
284 		int leapCenturyDays;
285 		int regularLeapDays;
286 		if (month <= 2) {
287 			year--;
288 			month += 12;
289 		} else {
290 			month++;
291 		}
292 
293 		// account for leap days using century
294 		// Pope Gregor's decree
295 		if (year > 1582 || year == 1582 && month > 10 || year == 1582
296 				&& month == 10 && dayOfMonth >= 15) {
297 			leapCenturyDays = 2 - year / 100 + year / 400;
298 		} else {
299 			leapCenturyDays = 0;
300 		}
301 
302 		// add in regular leap days
303 		if (year < 0) {
304 			regularLeapDays = (int) (365.25 * year - 0.75);
305 		} else {
306 			regularLeapDays = (int) (365.25 * year);
307 		}
308 
309 		final int D = (int) (30.6001 * (month + 1));
310 
311 		final double jdd = leapCenturyDays + regularLeapDays + D + dayOfMonth
312 				+ hour / 24 + minutes / (24 * 60) + seconds / (24 * 60 * 60)
313 				+ 1720994.5;
314 
315 		return jdd;
316 	}

420 		final Calendar cal = Calendar.getInstance();
421 		// default will be Gregorian in US Locales
422 		cal.setTime(endDate);
423 		final int endMonth = cal.get(Calendar.MONTH) + 1;
424 		final int endYear = cal.get(Calendar.YEAR);
425 		cal.setTime(startDate);
426 		final int startMonth = cal.get(Calendar.MONTH) + 1;
427 		final int startYear = cal.get(Calendar.YEAR);
428 
429 		if (logger.isDebugEnabled()) {
430 			logger.debug("Start Month is " + startMonth + " and Start Year is "
431 					+ startYear);
432 			logger.debug("End Month is " + endMonth + " and End Year is "
433 					+ endYear);
434 			logger.debug("Months is Year is " + cal.getMaximum(Calendar.MONTH));
435 		}
436 		// the following will work okay for Gregorian but will not
437 		// work correctly in a Calendar where the number of months
438 		// in a year is not constant
439 		return ((endYear - startYear) * (cal.getMaximum(Calendar.MONTH) + 1))
440 				+ (endMonth - startMonth);
441 	}
