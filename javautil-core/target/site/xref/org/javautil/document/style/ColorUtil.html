20  	/**
21  	 * Returns a CSS friendly color in #RRGGBB format, where RR, GG, and BB are
22  	 * the red, green, and blue values of the color in hex. A leading hash sign
23  	 * is always the first character.
24  	 * 
25  	 * @param color
26  	 * @return
27  	 */
28  	public static String toRgbHex(final Color color) {
29  		final StringBuffer hexString = new StringBuffer();
30  		hexString.append("#");
31  		hexString.append(Integer.toHexString(color.getRed()));
32  		if (hexString.length() < 3) {
33  			hexString.append("0");
34  		}
35  		hexString.append(Integer.toHexString(color.getGreen()));
36  		if (hexString.length() < 5) {
37  			hexString.append("0");
38  		}
39  		hexString.append(Integer.toHexString(color.getBlue()));
40  		if (hexString.length() < 7) {
41  			hexString.append("0");
42  		}
43  
44  		return hexString.toString();
45  	}

47  	/**
48  	 * Returns a java.awt.Color by reading the RGB values from a hexadecimal
49  	 * string of format #RRGGBB (the same format used by toRgbHex). An
50  	 * IllegalArgumentException is thrown if the string format is not valid.
51  	 * 
52  	 * @param rgbHexString
53  	 * @return
54  	 */
55  	public static Color parseColorFromRgbHexString(final String rgbHexString) {
56  		String t = rgbHexString;
57  		Color ret = null;
58  		if (t.charAt(0) != '#') {
59  			throw new IllegalArgumentException(
60  					"color RGB value must begin with hash (#) mark");
61  		}
62  		if (t.length() == 4) {
63  			// convert three hex value to six hex value
64  			t = new String(new char[] { t.charAt(0), t.charAt(1), t.charAt(1),
65  					t.charAt(2), t.charAt(2), t.charAt(3), t.charAt(3) });
66  		}
67  		if (t.length() != 7) {
68  			throw new IllegalArgumentException("format expected to be "
69  					+ "in the format #RGB or #RRGGBB; " + "format was "
70  					+ rgbHexString);
71  		}
72  		// parse as six hex value
73  		int rValue, gValue, bValue = 0;
74  		try {
75  			rValue = Integer.parseInt(t.substring(1, 3), 16);
76  			gValue = Integer.parseInt(t.substring(3, 5), 16);
77  			bValue = Integer.parseInt(t.substring(5, 7), 16);
78  			ret = new Color(rValue, gValue, bValue);
79  		} catch (final Exception e) {
80  			throw new IllegalArgumentException("bad RGB hex format");
81  		}
82  		return ret;
83  	}

86  	 * Convenience method for reading a string as a java.awt.Color.
87  	 * 
88  	 * If the color starts with hash sign, the color is parsed using the method
89  	 * parseColorFromRgbHexString,
90  	 * 
91  	 * todo jjs where is this suitable color name lookup? otherwise a suitable
92  	 * color name lookup is used.
93  	 * 
94  	 * @param color
95  	 * @return
96  	 */
97  	public static Color parseColor(final String color) {
98  		Color ret = null;
99  		if (color.toLowerCase().equals("transparent")) {
100 			ret = TRANSPARENT;
101 		} else if (color.charAt(0) == '#') {
102 			ret = parseColorFromRgbHexString(color);
103 		} else {
104 			throw new IllegalArgumentException("unknown color string: " + color);
105 		}
106 		return ret;
107 	}
