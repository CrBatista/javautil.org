119 		settings[0] = setting;

121 		settings[2] = setting;
122 		settings[3] = setting;
123 		for (int i = 0; i < 4; i++) {

149 		String[] args = new String[0];
150 		if (lineSettings != null) {
151 			args = lineSettings.split(" ");
152 		}
153 		Color lineColor = null;
154 		Integer lineWidth = 0;
155 		try {
156 			if (args.length > 0) {
157 				lineWidth = Integer.parseInt(args[0].trim());
158 			}
159 			if (args.length > 1) {
160 				lineColor = ColorUtil.parseColor(args[1].trim());
161 			} else if (args.length > 0) {
162 				lineColor = Color.BLACK;
163 			}
164 		} catch (final Exception e) {
165 			throw new RuntimeException("unable to parse as simple border: "
166 					+ lineSettings, e);
167 		}
168 		return new LineSettings(lineWidth, lineColor);
