64  	public HSSFCell addCell(final int rowIndex, final int columnIndex,
65  			final Object value, final Style style) {
66  
67  		final HSSFCell cell = getCell(rowIndex, columnIndex);
68  		if (value != null) {
69  			if (value instanceof Number) {
70  				cell.setCellValue(((Number) value).doubleValue());
71  			} else if (value instanceof Date) {
72  				cell.setCellValue((Date) value);
73  			} else if (value instanceof String) {
74  				String val = (String) value;
75  				if (((String) value).indexOf("\\n") > -1) {
76  					val = val.replace("\\n", "\n");
77  				}
78  				cell.setCellValue(new HSSFRichTextString(val));
79  			} else if (value instanceof Boolean) {
80  				final Boolean bool = (Boolean) value;
81  				cell.setCellValue(bool.booleanValue());
82  			}
83  		} else {
84  			cell.setCellValue("");
85  		}
86  		if (style != null) {
87  			cell.setCellStyle(styleFactory.getHSSFCellStyle(style));
88  		}
89  		return cell;
90  	}

143 	public HSSFCell addColumnFormulaCell(final int rowIndex,
144 			final int columnIndex, final int firstDataRowIndex,
145 			final int lastDataRowIndex, final Style style, final String function) {
146 
147 		if (logger.isDebugEnabled()) {
148 			final String message = "addColumnFormulaCell " + rowIndex + " "
149 					+ columnIndex + " " + firstDataRowIndex + " "
150 					+ lastDataRowIndex + " " + function;
151 			logger.debug(message);
152 		}
153 		final String firstAddress = CellAddress.getCellAddress(
154 				firstDataRowIndex, columnIndex);
155 		final String secondAddress = CellAddress.getCellAddress(
156 				lastDataRowIndex, columnIndex);
157 		final String formula = function + "(" + firstAddress + ":"
158 				+ secondAddress + ")";
159 		final HSSFRow row = getRow(rowIndex);
160 		final HSSFCell cell = row.createCell(columnIndex);
161 		cell.setCellFormula(formula);
162 		final HSSFCellStyle hstyle = styleFactory.getHSSFCellStyle(style);
163 		cell.setCellStyle(hstyle);
164 		return cell;

192 	public void fixColumnWidth(final int columnIndex) {
193 		final int rowCount = sheet.getPhysicalNumberOfRows();
194 		int maxWidth = 0;
195 		for (int i = 0; i < rowCount; i++) {
196 			final HSSFRow r = sheet.getRow(i);
197 			final HSSFCell c = r.getCell(columnIndex);
198 
199 			if (c != null) {
200 				final String v = c.getStringCellValue();
201 				if (v != null && v.length() > maxWidth) {
202 					maxWidth = v.length();
203 				}
204 			}
205 		}
206 		if (maxWidth > 8) {
207 			sheet.setColumnWidth(columnIndex, maxWidth);
208 		}
209 	}

211 	public void emitRegion(final Dataset dataset,
212 			final List<ColumnMetadata> columns, final int startingRow,
213 			final int startingColumn) {
214 		if (logger.isDebugEnabled()) {
215 			final StringBuilder b = new StringBuilder();
216 			for (final ColumnMetadata column : columns) {
217 				b.append(column);
218 				b.append(SystemProperties.newline);
219 			}
220 			b.append("starting row " + startingRow + SystemProperties.newline);
221 			b.append("startingColumn " + startingColumn
222 					+ SystemProperties.newline);
223 
224 			b.append(DatasetMarshaller.asString(dataset));
225 			logger.debug(b.toString());
226 
227 		}
228 		final int dataRows = dataset.getDatasetIterator().getRowCount();
229 		if (startingRow + dataRows + 2 > MAX_ROWS_PER_SHEET) {
230 			throw new IllegalArgumentException(
231 					"would produce too many rows in sheet: " + dataRows
232 							+ " startingRow " + startingRow);
233 		}
234 		emitHeaders(columns, styleFactory.getBaseHeaderStyle(), startingRow,
235 				startingColumn);
236 		final int lastDataRowIndex = emitData(dataset, columns,
237 				styleFactory.getBaseDataStyle(), startingRow + 1,
238 				startingColumn);
239 		emitSummary(columns, lastDataRowIndex + 1, startingColumn,
240 				startingRow + 1, lastDataRowIndex,
241 				styleFactory.getBaseFooterStyle());
242 	}

250 	public void emitHeaders(final List<ColumnMetadata> columns,
251 			final Style baseStyle, final int startingRow,
252 			final int startingColumn) {
253 
254 		if (baseStyle == null) {
255 			throw new IllegalArgumentException("baseStyle is null");
256 		}
257 		final int rowIndex = startingRow;
258 		int columnIndex = startingColumn;
259 		int entry = 0;
260 		for (final ColumnMetadata column : columns) {
261 			Style cellStyle = null;
262 			String heading = "";
263 			if (column == null) {
264 				logger.warn("null column at entry " + entry);
265 				cellStyle = baseStyle;
266 			} else {
267 				cellStyle = getStyle(baseStyle, null,
268 						column.getHorizontalAlignment());
269 				heading = column.getHeading();
270 			}
271 			if (heading == null) {
272 				heading = column.getColumnName();
273 			}
274 			addCell(rowIndex, columnIndex, heading, cellStyle);
275 			setColumnWidth(columnIndex, getColumnWidth(column));
276 
277 			columnIndex++;
278 			entry++;
279 		}

305 	protected int emitData(final Dataset dataset,
306 			final List<ColumnMetadata> columns, final Style baseStyle,
307 			final int startingRow, final int startingColumn) {
308 		if (dataset == null) {
309 			throw new IllegalArgumentException("dataset is null");
310 		}
311 		if (columns == null) {
312 			throw new IllegalArgumentException("columns is null");
313 		}
314 		if (baseStyle == null) {
315 			throw new IllegalArgumentException("baseStyle is null");
316 		}
317 		final Map<String, Integer> columnIndexByName = ColumnMetaMap
318 				.getColumnNameIndexMap(columns);
319 
320 		final DatasetIterator datasetIterator = dataset.getDatasetIterator();
321 		int rowIndex = startingRow;
322 		while (datasetIterator.next()) {
323 			if (logger.isTraceEnabled()) {
324 				logger.trace("adding row " + rowIndex);
325 			}
326 			int columnIndex = startingColumn;
327 			for (final ColumnMetadata column : columns) {
328 				if (column == null) {
329 					throw new IllegalStateException("column is null");
330 				}
331 
332 				final Style cellStyle = getStyle(baseStyle, column);
333 
334 				if (column.getWorkbookFormula() == null) {
335 					final Object data = datasetIterator.getObject(column
336 							.getColumnName());
337 
338 					addCell(rowIndex, columnIndex, data, cellStyle);
339 					if (logger.isDebugEnabled()) {
340 						final String message = "adding column " + rowIndex
341 								+ " " + columnIndex + " " + data + " "
342 								+ cellStyle + " " + column.toString();
343 						logger.debug(message);
344 					}
345 				} else {
346 					final HSSFCell cell = addFormulaCell(
347 							column.getWorkbookFormula(), columnIndexByName,
348 							column.getColumnName(), rowIndex, columnIndex);
349 					if (logger.isDebugEnabled()) {
350 						final String message = "adding formua " + rowIndex
351 								+ " " + columnIndex + " " + cellStyle + " "
352 								+ column.toString();
353 						logger.debug(message);
354 					}
355 					cell.setCellStyle(styleFactory.getHSSFCellStyle(cellStyle));
356 				}
357 				columnIndex++;
358 			}
359 			rowIndex++;
360 		}
361 		return rowIndex - 1;
362 	}
