194 	public void setNullable(final String val) {
195 		if ("N".equals(val)) {
196 			nullable = Boolean.FALSE;
197 		} else if ("Y".equals(val)) {
198 			nullable = Boolean.TRUE;
199 		} else if (XmlMeta.NULL_ALLOWED.equals(val)) {
200 			nullable = Boolean.TRUE;
201 		} else if (XmlMeta.NOT_NULL.equals(val)) {
202 			nullable = Boolean.FALSE;
203 		} else if (val == null) {
204 			nullable = null;
205 		} else {
206 			throw new IllegalArgumentException("expected '"
207 					+ XmlMeta.NULL_ALLOWED + "' |  '" + XmlMeta.NOT_NULL
208 					+ "' | null | 'Y' | 'N' but got '" + val + "'");
209 		}
