91  	@Override
92  	protected ModelAndView handleRequestInternal(HttpServletRequest request,
93  			HttpServletResponse response) throws Exception {
94  		// get the remote url, with all desired modifications
95  		URL remoteUrl = buildRemoteURL(request);
96  
97  		if (logger.isDebugEnabled()) {
98  			logger.debug("remoteUrl is " + remoteUrl.toExternalForm());
99  		}
100 
101 		// establish remote connection
102 		URLConnection connection = remoteUrl.openConnection();
103 		boolean isHttpConnection = HttpURLConnection.class
104 				.isAssignableFrom(HttpURLConnection.class);
105 
106 		// proxy http/https headers
107 		if (proxyHttpRequestHeaders && isHttpConnection) {
108 			HttpURLConnection httpConnection = (HttpURLConnection) connection;
109 			proxyHttpHeaders(httpConnection, response);
110 		}
111 		boolean isHttpSuccess = isHttpConnection
112 				&& ((HttpURLConnection) connection).getResponseCode() % 200 < 100;
113 
114 		// send the body of the proxied content
115 		if (!isHttpConnection || isHttpSuccess) {
116 			IOUtils.pump(connection.getInputStream(), response
117 					.getOutputStream());
118 		}
119 
120 		// we don't need a model and view, as the response was already written
121 		return null;
122 	}

133 	/**
134 	 * Returns the desired a remote url, suitable for proxying.
135 	 * 
136 	 * @param request
137 	 * @return url
138 	 * @throws MalformedURLException
139 	 */
140 	public URL buildRemoteURL(HttpServletRequest request)
141 			throws MalformedURLException {
142 
143 		StringBuilder remoteAddress = new StringBuilder(remoteUri);
144 
145 		// append the actual requested url to the proxied request
146 		if (proxyRequestUri) {
147 			if (isAddTrailingSlash()
148 					&& remoteAddress.charAt(remoteAddress.length() - 1) != '/') {
149 				remoteAddress.append("/");
150 			}
151 			String localUri = request.getRequestURI();
152 			// remove the stripPrefix if specified
153 			if (stripPrefix != null && localUri.startsWith(stripPrefix)) {
154 				localUri = localUri.substring(stripPrefix.length());
155 			}
156 			remoteAddress.append(localUri);
157 		}
158 
159 		Map<String, List<String>> parameters = new LinkedHashMap<String, List<String>>();
160 		// append custom parameters
161 		if (additionalParameters != null) {
162 			appendCustomParameters(parameters);
163 		}
164 		// append local parameters, do not allow overriding of a custom
165 		// parameters
166 		if (proxyRequestParameters) {
167 			appendLocalUrlParameters(parameters, request);
168 		}
169 
170 		// add the build parameter map to the url as request parameters
171 		try {
172 			if (parameters.keySet().size() > 0) {
173 				boolean needsQ = remoteAddress.indexOf("?") == -1;
174 				for (String parmName : parameters.keySet()) {
175 					List<String> values = parameters.get(parmName);
176 					if (values.size() == 0) {
177 						values = new ArrayList<String>();
178 						values.add("");
179 					}
180 					for (String value : values) {
181 						remoteAddress.append(needsQ ? "?" : "&");
182 						remoteAddress.append(parmName);
183 						if (value != null && !"".equals(value)) {
184 							remoteAddress.append("=");
185 							remoteAddress.append(URLEncoder.encode(value,
186 									"UTF-8"));
187 						}
188 						needsQ = false;
189 					}
190 				}
191 			}
192 		} catch (Exception e) {
193 			throw new IllegalStateException("error while bulding url", e);
194 		}
195 
196 		URL remoteUrl = new URL(remoteAddress.toString());
197 		return remoteUrl;
198 	}

200 	/**
201 	 * Reads the parameters from the request and appends them to the parameter
202 	 * map.
203 	 * 
204 	 * @param parameters
205 	 * @param request
206 	 */
207 	protected void appendLocalUrlParameters(
208 			Map<String, List<String>> parameters, HttpServletRequest request) {
209 		Map<String, List<String>> urlParams = HttpRequestUtils
210 				.buildParameterMap(request);
211 		for (String parameterName : urlParams.keySet()) {
212 			boolean isOverriden = additionalParameters != null
213 					&& additionalParameters.containsKey(parameterName);
214 			boolean isIgnored = stripParameterNames != null
215 					&& stripParameterNames.contains(parameterName);
216 			if (!isOverriden && !isIgnored) {
217 				parameters.put(parameterName, urlParams.get(parameterName));
218 			}
219 		}
220 	}

222 	/**
223 	 * Adds the custom parameters specified as bean properties to the parameter
224 	 * map.
225 	 * 
226 	 * @param parameters
227 	 */
228 	@SuppressWarnings("unchecked")
229 	protected void appendCustomParameters(Map<String, List<String>> parameters) {
230 		for (String parameterName : additionalParameters.keySet()) {
231 			Object rawParameter = additionalParameters.get(parameterName);
232 			List<String> parameterValues = new ArrayList<String>();
233 			if (rawParameter != null) {
234 				if (Collection.class.isAssignableFrom(rawParameter.getClass())) {
235 					Collection<Object> collection = (Collection<Object>) rawParameter;
236 					for (Object objectValue : collection) {
237 						parameterValues.add(String.valueOf(objectValue));
238 					}
239 				} else {
240 					parameterValues.add(String.valueOf(rawParameter));
241 				}
242 			}
243 			parameters.put(parameterName, parameterValues);
244 		}
245 	}

247 	/**
248 	 * Proxies http headers from a connection to another response.
249 	 * 
250 	 * @param connection
251 	 * @param response
252 	 * @throws IOException
253 	 */
254 	protected void proxyHttpHeaders(HttpURLConnection connection,
255 			HttpServletResponse response) throws IOException {
256 		int responseCode = 500;
257 		try {
258 			responseCode = connection.getResponseCode();
259 		} catch (IOException e) {
260 			logger.error(e);
261 		}
262 		response.setStatus(responseCode);
263 		int last = connection.getHeaderFields().size();
264 		for (int i = 0; i < last; i++) {
265 			String name = connection.getHeaderFieldKey(i);
266 			String value = connection.getHeaderField(name);
267 			if (name == null && value == null) {
268 				break;
269 			}
270 			if (name != null) {
271 				response.setHeader(name, value);
272 			}
273 		}
274 	}
