22  	 * Translates the request parameters into a map that contains a list of
23  	 * values for each key.
24  	 * 
25  	 * @param request
26  	 * @return map
27  	 */
28  	@SuppressWarnings("unchecked")
29  	public static Map<String, List<String>> buildParameterMap(
30  			HttpServletRequest request) {
31  		// accumulate our parameter map of values
32  		LinkedHashMap<String, List<String>> requestParameters = new LinkedHashMap<String, List<String>>();
33  		Map<String, Object> _params = request.getParameterMap();
34  		for (String paramName : _params.keySet()) {
35  			Object _paramValue = _params.get(paramName);
36  			ArrayList<String> paramValues = null;
37  			if (_paramValue instanceof String) {
38  				String paramValue = (String) _paramValue;
39  				paramValues = new ArrayList<String>(1);
40  				paramValues.add(paramValue);
41  			} else if (_paramValue instanceof String[]) {
42  				String[] paramValue = (String[]) _paramValue;
43  				paramValues = new ArrayList<String>(paramValue.length);
44  				for (int i = 0; i < paramValue.length; i++) {
45  					paramValues.add(paramValue[i]);
46  				}
47  			}
48  			requestParameters.put(paramName, paramValues);
49  		}
50  		return requestParameters;
51  	}

69  	 * prefix is matched case-sensitive. Optionally the prefix can be stripped
70  	 * in the returned map. The returned map is a map containing the request
71  	 * parameter name and values as as a list.
72  	 * 
73  	 * @param prefix
74  	 * @param request
75  	 * @param stripPrefix
76  	 * @return map of parameters names/values list
77  	 */
78  	public static Map<String, List<String>> getParametersLike(String prefix,
79  			HttpServletRequest request, boolean stripPrefix) {
80  		Map<String, List<String>> parms = buildParameterMap(request);
81  		Map<String, List<String>> matches = new LinkedHashMap<String, List<String>>();
82  		for (String parmName : parms.keySet()) {
83  			if (parmName.startsWith(prefix)) {
84  				String parm = parmName;
85  				if (stripPrefix) {
86  					parm = parm.substring(parmName.length());
87  				}
88  				matches.put(parm, parms.get(parmName));
89  			}
90  		}
91  		return matches;
92  	}

95  	 * Gets parameters from the http request that start with a given prefix. The
96  	 * prefix is matched case-sensitive. Optionally the prefix can be stripped
97  	 * in the returned map. The returned map is a map containing the request
98  	 * parameter name and value. Fails fast when a parameter is repeated and
99  	 * starts with the prefix specified.
100 	 * 
101 	 * @param prefix
102 	 * @param request
103 	 * @param stripPrefix
104 	 * @return map of parameters names/values
105 	 */
106 	public static Map<String, String> getSingleParametersLike(String prefix,
107 			HttpServletRequest request, boolean stripPrefix) {
108 		Map<String, List<String>> parms = buildParameterMap(request);
109 		Map<String, String> matches = new LinkedHashMap<String, String>();
110 		for (String parmName : parms.keySet()) {
111 			if (parmName.startsWith(prefix)) {
112 				String parm = parmName;
113 				if (stripPrefix) {
114 					parm = parm.substring(prefix.length());
115 				}
116 				List<String> values = parms.get(parmName);
117 				if (values.size() > 1) {
118 					throw new IllegalStateException("parameter " + parmName
119 							+ " is present in the http request values "
120 							+ values.size() + " times, but was expected "
121 							+ "to appear at most once; values: "
122 							+ Arrays.toString(values.toArray()));
123 				}
124 				String value = values.size() == 0 ? null : values.get(0);
125 				matches.put(parm, value);
126 			}
127 		}
128 		return matches;
129 	}

158 	 * Gets the URI path and parameters for a given request.
159 	 * 
160 	 * @param request
161 	 * @return path
162 	 */
163 	public static String getRequestedPage(HttpServletRequest request) {
164 		// store the full URI and query for later reference
165 		String queryString = request.getQueryString();
166 		String page = request.getRequestURI();
167 		if (queryString != null && queryString.length() > 0) {
168 			page = request.getRequestURI() + "?" + queryString;
169 		} else {
170 			page = request.getRequestURI();
171 		}
172 		while (page.length() > 2 && page.charAt(0) == '/'
173 				&& page.charAt(1) == '/') {
174 			// tomcat 5.5 can have an incorrect double leading slash, remove it
175 			page = page.substring(1, page.length());
176 		}
177 		String contextPath = request.getContextPath();
178 		if (contextPath.length() > 0 && page.startsWith(contextPath)) {
179 			page = page.substring(contextPath.length(), page.length());
180 		}
181 		return page;
182 	}
