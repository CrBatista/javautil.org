67  	public String map(final String in) {

70  		}
71  
72  		String find = in.toUpperCase().trim();
73  		find = find.replace(".", "");
74  		// logger.info("looking for '" + find + "'");
75  		final String found = directionMap.get(find);
76  		if (logMisses) {
77  			logMiss(find);
78  		}
79  		return found;
80  	}
