152 	@Override
153 	public synchronized void standardize(final Address raw,
154 			final StandardizedAddress std)
155 			throws AddressStandardizationException {
156 		if (raw == null) {
157 			throw new IllegalArgumentException("raw is null");
158 		}
159 
160 		this.raw = raw;
161 		this.std = std;
162 		exceptions = new ArrayList<AddressStandardizationException>();
163 		final ZipCodeFormatter zipFormatter = new ZipCodeFormatter();
164 		addr1 = raw.getAddress1();
165 		addr2 = raw.getAddress2();
166 		try {
167 			// these must occur before process address lines so subunits and po
168 			// boxes can be stripped
169 			processPoBox();
170 			processSubunit();
171 			//
172 			processAddressLines();
173 			processRuralRoute();
174 			std.setCity(cityStandardizer.getStandardCityName(raw.getCity(),
175 					std.getState(), null)); // raw.getRawCountryCode()));
176 			std.setState(StateAbbreviations.getStandardAbbreviation(raw
177 					.getState()));
178 			std.setCountryCode(raw.getCountryCode() == null ? null : raw
179 					.getCountryCode().trim().toUpperCase());
180 			checkResults();
181 			std.setPostalCode(zipFormatter.formatUSPostalCode(raw
182 					.getPostalCode()));
183 		} catch (final AddressStandardizationException e) {
184 			exceptions.add(e);
185 			throw e;
186 		}
187 
188 	}

190 	/**
191 	 * <p>processAddressLines.</p>
192 	 *
193 	 * @throws org.javautil.address.exception.AddressStandardizationException if any.
194 	 */
195 	protected void processAddressLines() throws AddressStandardizationException {
196 
197 		final boolean addr1HasStreet = streetLine1Parser.parse(addr1);
198 		final boolean addr2HasStreet = streetLine2Parser.parse(addr2);
199 
200 		if (addr1HasStreet && addr2HasStreet) {
201 			throw new MultipleStreetAddressesException(raw);
202 		}
203 
204 		String streetAddress = null;
205 		if (addr1HasStreet || addr2HasStreet) {
206 			final StreetParser sp = addr1HasStreet ? streetLine1Parser
207 					: streetLine2Parser;
208 			std.setStreetNumber(sp.getStreetNumber());
209 			std.setStreetName(sp.getStreetName());
210 			std.setStreetType(sp.getStreetType());
211 			streetAddress = streetAddressFormatter.format(sp);
212 			if (streetAddress.length() > maxStdStreetAddrLength) {
213 				final String message = "street address length for '"
214 						+ streetAddress + "' length " + streetAddress.length()
215 						+ " exceeds limit of " + maxStdStreetAddrLength;
216 				throw new StandardizedAddressLineWouldBeTooLongException(raw,
217 						message);
218 			}
219 			logger.debug("streetAddress: '" + streetAddress + "'");
220 			setStreetAddressLine(streetAddress);
221 		}
222 		if (logger.isDebugEnabled()) {
223 			final StringBuilder sb = new StringBuilder();
224 			sb.append("addr1: ").append(addr1).append(newline);
225 			sb.append("addr2: ").append(addr2).append(newline);
226 			sb.append("streetAddress: '").append(streetAddress).append("'");
227 			logger.debug(sb.toString());
228 		}
229 	}

236 	protected void processPoBox() throws AddressStandardizationException {
237 
238 		final boolean addr1HasPoBox = poBoxLine1Parser.parse(addr1);
239 		final boolean addr2HasPoBox = poBoxLine2Parser.parse(addr2);
240 		// todo should throw an exception if it has a PO Box and a street
241 		// address
242 		if (addr1HasPoBox && addr2HasPoBox) {
243 			throw new MultiplePoBoxAddressException(raw);
244 		}
245 
246 		if (addr1HasPoBox || addr2HasPoBox) {
247 			final POBoxParser sp = addr1HasPoBox ? poBoxLine1Parser
248 					: poBoxLine2Parser;
249 			std.setPoBox(sp.getPoBox());
250 			setStreetAddressLine(poBoxFormatter.format(sp));
251 		}

254 	/**
255 	 * <p>processRuralRoute.</p>
256 	 *
257 	 * @throws org.javautil.address.exception.AddressStandardizationException if any.
258 	 */
259 	protected void processRuralRoute() throws AddressStandardizationException {
260 		final boolean addr1Route = routeLine1Parser.parse(addr1);
261 		final boolean addr2Route = routeLine2Parser.parse(addr2);
262 		if (addr1Route && addr2Route) {
263 			throw new MultipleRuralRoutesException(raw);
264 		}
265 
266 		if (addr1Route || addr2Route) {
267 			if (std.getStreetNumber() != null) {
268 				throw new RuralRouteAndStreetAddressException(raw);
269 
270 			}
271 
272 			final RuralRouteParser routeParser = addr1Route ? routeLine1Parser
273 					: routeLine2Parser;
274 			final String addrLine = "RR " + routeParser.getRuralRoute()
275 					+ " Box " + routeParser.getBox();
276 			setStreetAddressLine(addrLine);
277 		}
278 	}

280 	/**
281 	 * <p>processSubunit.</p>
282 	 *
283 	 * @throws org.javautil.address.exception.AddressStandardizationException if any.
284 	 */
285 	protected void processSubunit() throws AddressStandardizationException {
286 
287 		final boolean addr1HasSubunit = subunitLine1Parser.parse(addr1);
288 		final boolean addr2HasSubunit = subunitLine2Parser.parse(addr2);
289 
290 		if (addr1HasSubunit && addr2HasSubunit) {
291 			throw new MultipleSubunitException(raw);
292 		}
293 
294 		if (addr1HasSubunit) {
295 			addr1 = subunitLine1Parser.getNonUnit();
296 		}
297 		if (addr2HasSubunit) {
298 			addr2 = subunitLine2Parser.getNonUnit();
299 		}
300 
301 		if (addr1HasSubunit || addr2HasSubunit) {
302 			final SubunitParser sp = addr1HasSubunit ? subunitLine1Parser
303 					: subunitLine2Parser;
304 			std.setSubunitType(sp.getSubunitType());
305 			std.setSubunitCode(sp.getSubunitCode());
306 			final String subunit = sp.getSubunitTypeStandardName() + " "
307 					+ sp.getSubunitCode();
308 			std.setAddress1(subunit);
309 		}
310 	}
